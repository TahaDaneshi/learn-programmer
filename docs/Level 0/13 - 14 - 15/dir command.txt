dir =>  فهرست دایرکتوری که روش تنظیم شده ایم

dir /? => راهنمایی کامند 

dir [<drive :>][<path>][<file name>] =>فهرست گیری از پوشه مشخص شده 

dir [<file name>] => فهرست گیری از فایلها با فرمت مشخص شده

dir /p => فهرست گیری در ابعاد پنجره کامند پرامت

dir /q => نشان دادن صاحب فایل ها (مالکیت )

dir /w => نشان دادن فهرست به صورت خطی و سطری (8 نام در هر سطر )

dir /d =>نشان دادن فهرست به صورت خطی و سطری (6 نام در هر سطر )

========================================================================================

dir /a => نشان دادن فهرست فایل ها با ویژگی های خاص  (اگر این پارامتر را استفاده نکنید همه ی فایل ها به غیر از فایل های مخفی و سیستمی نشان داده میشوند)

<<<<<<< خود این دستور دارای چندین پارامتر است  >>>>>>>
d - دایرکتوری ها
h - فایل های مخفی 
s - فایل های سیستمی
l - فایل های امتیازی
r - فایل های با قابلیت read-only
a - فایل های که اماده بایگانی شدن هستند
i - پرونده های ایندکس گزاری نشده

<<<<<< این فایل ها به صورت زیر به کار میروند >>>>>>
dir /ad
dir /ah
dir /as
dir /al
dir /ar
dir /aa
dir /ai
...................
dir /a-s  برای نمایش ندادن فایل های سیستمی
dir /ah-s 
...................
dir /ahs برای نمایش فایل با چندین ویژگی 
...................
dir /a:d  => دستورات بالا به شکل زیر نیز به کار میروند
========================================================================================


dir /o =>نشان دادن فهرست با شخصی سازی ( بر اساس :سایز،زمان  و..)

<<<<<<<<<<<خود این دستور همانند دستور بالا دارای چندین پارامتر است>>>>>>>>>>>>

n - مرتب سازی بر اساس حروف الفبا (نام)
e - مرتب سازی بر اساس حروف الفبا (افزونه)
g - مرتب سازی بر اساس گروه (ابتدا پوشه ها)
s - مرتب سازی بر اساس سایز (کوچک اول)
d - مرتب سازی بر اساس تاریخ و زمان (قدیمی تر اول )
برای معکوس کردن ترتیب مرتب سازی از پیشوند (-)استفاده کنید

<<<<<<<<< به صورت زیر به کار میروند >>>>>>>>>

dir /on
dir /oe
dir /og
dir /os
dir /od

...............................
dir /o-s => مرتب سازی بر اساس سایز (بزرگ  اول)
...............................
dir /osd => فهرستی که فایل های قدیمی و کوچک در ان اول هستند
...............................
dir /o:s => به این صورت نیز به کار میروند 
========================================================================================
dir /t => کنترل می کند که کدام قسمت زمانی برای مرتب سازی نمایش داده شده یا استفاده شده است

<<<<<<<<<<<<<<<<< این دستور دارای 3 پارامتر میباشد >>>>>>>>>>>>>>>>>>>>>

c - زمانی که ایجاد شده
a - اخرین دسترسی
w - اخرین ویرایش

========================================================================================
dir /s => تمامی زیر شاخه ها را لیست میکند

dir /b =>فهرستی لخت بدون اطلاعات اضافی

dir /l => نمایش فایل ها با نام کوچک 

dir /n  =>نمایش نام ها در سمت راست و تاریخ در سمت چپ 
dir /-n =>نمایش نام ها در سمت چپ و تاریخ در سمت راست 

dir /x => نمایش نام کوتاه و بزرگ در کنار هم

dir /? => کمک
